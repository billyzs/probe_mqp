1classdef MVPDriver < MotorDriver
  2    properties (Access=protected)
  3        name = 'MVP';
  4        comPort; % COM1, COM2, ...
  5        address; % Sequence 1, 2 , 3 ... 
  6        defaultVelocity = 100;
  7        defaultAcceleration = 100;
  8        maxDisplacement = 40000;
  9        maxVelocity = 4000;
 10        maxAcceleration = 4000;
 11        displacement;
 12        moves = zeros(1,3); % moves is a n*3 Array (Stack) containing the past moves

 14    
 15    methods (Access=private)
 16        function connect(this)
 17            this.writeCmd('HO');
 18        end
 19        function enableMotor(this)
 20            this.writeCmd('EN');
 21        end
 22        function cmd = parseCommand(this, c, val)
 23            if nargin < 3 || strcmp(val, '')
 24                val = '';
 25            else 
 26                val = [' ', int2str(val)];
 27            end
 28            cmd = [int2str(this.address), ' ', c, val];
 29        end
 30    end
 31    
 32    methods (Access=public)
 33   
 34        function enable(this)
 35            try
 36                fopen(this.comPort);
 37                this.connect();
 38                this.enableMotor();
 39                this.writeCmd('ANO', 2350); % ANO sets the current. has to be set to 2350 per NA rep Jim Steward
 40            catch exception
 41                fclose(this.comPort);
 42                clear obj;
 43                rethrow(exception);
 44            end
 45        end
 46        function disable(this)
 47            if (this.isConnected())
 48                fclose(this.comPort);
 49            end
 50        end
 51        function success = doMove(this, displacement, vel, accel)
 52            success = 0;
 53            try
 54                if nargin == 4
 55                    prefix = {'LA', 'SP', 'AC'};
 56                    cmd = [displacement, vel, accel];
 57                elseif nargin == 3
 58                    prefix = {'LA', 'SP'};
 59                    cmd = [displacement, vel];
 60                else
 61                    prefix = {'LA'};
 62                    cmd = [displacement];
 63                end
 64
 65                % parse & write command
 66
 67                for t = 1:(nargin-1)
 68                    this.writeCmd(char(prefix(t)), cmd(t));
 69                end
 70                % write move cmd and get status;
 71                pause(0.2); %% definitely need this pause...
 72                this.writeCmd('M');
 73
 74                % this.writeCmd('M');
 75                % pos = fscanf(this.comPort);
 76                % this.writeCmd('ST');
 77                % status = fscanf(this.comPort);
 78                success = 1;
 79            catch exception
 80                rethrow(exception);
 81            end
 82        end
 83        
 84    end
 85    methods (Access=public)
 86        % constructor

 87        function obj = MVPDriver(addr, port)
 88            % the addr specified by the switches on the actual controller
 89            obj.address = addr; 
 90            if nargin > 1
 91                try
 92                    obj.comPort = serial(port,'BaudRate', 9600, 'DataBits',8,  'StopBits', 1,... 
 93                        'Parity', 'none','FlowControl', 'none', ...
 94                        'ReadAsyncMode', 'continuous', ...
 95                        'Terminator', 'CR', ...
 96                        'ByteOrder', 'littleEndian');    
 97                catch ME
 98                    fclose(obj.comPort);
 99                    clear obj;
100                    rethrow(ME);            
101                end
102            end
103        end
104        

105        function closePort(this)
106            if (~isempty(this.comPort) && strcmp(this.comPort.Status,'open'))
107                fclose(this.comPort);
108            end
109        end
110        
111        function success = isConnected(this)
112            success = false;
113            try
114                if( ~isempty(this.comPort) && strcmp(this.comPort.Status,'open'))
115                    success = true;
116                end
117            catch exception
118                rethrow(exception);
119            end
120        end
121        
122        function writeCmd(this, cmd, val)
123            if nargin < 3
124                val = '';
125            end
126            cmd = this.parseCommand(cmd, val);
127            try
128                disp(cmd);
129                assert(this.isConnected());
130                fprintf(this.comPort, cmd);
131                %pause(0.1);
132            catch ME
133                mymsg = ['Error writing command: ' cmd ' to serial port. Port is ' this.comPort.Status];
134                disp(mymsg);
135                % cause = MException('AcctError:NoClient', mymsg);
136                % ME = addCause(ME, cause);
137                rethrow(ME);
138            end
139        end
140        function cp = getComPort(this)
141            cp = this.comPort;
142        end
