 29        function valid = isMoveValid(this, displacement, velocity, acceleration)
 30            valid = 0;
 31            if ( ...
 32                     (velocity < this.maxVelocity)... 
 33                    )
 34                % do something
 35                valid = 1;
 36            else
 37                errMsg = [this.name,  '::isMoveValid::'];
 38                if (this.displacement + displacement > this.maxDisplacement)
 39                    errMsg = [errMsg  'displacement exceeds maxDisplacement: would be ', ...
 40                        + num2str(this.displacement + displacement)] + ';'; 
 41                end
 42                if (velocity > this.maxVelocity)
 43                     errMsg = [errMsg, 'vel exceeds maxVelocity: ', ...
 44                         num2str(velocity), ' vs ',  num2str(this.maxVelocity)  ';'];
 45                end
 46                if (acceleration > this.maxAcceleration)
 47                     errMsg = [errMsg  'acc exceeds maxAcceleration: ' ...
 48                         num2str(acceleration) ' vs ' num2str(this.maxAcceleration) ';'];
 49                end
 50                error(errMsg);
 51            end
 52        end
 53        
 54        function success = move(this, displacement, velocity, acceleration)
 55            success = 0;
 56            try
 57                if(true == isMoveValid(this, displacement, velocity, acceleration))
 58                    success = doMove(this, displacement, velocity, acceleration);
 59                    if success
 60                        this.moves = [displacement, velocity, acceleration; this.moves]; %LIFO
 61                        this.displacement = this.displacement + displacement;
 62                    end
 63                    return
 64                end
 65            catch exception
 66                rethrow(exception);
 67            end
 68        end
 69        function success = defaultMove(this, displacement)
 70            success = move(this, displacement, this.defaultVelocity, this.defaultAcceleration);
 71        end
 72        
 73        function success = undoLastMove(this)
 74            success = doMove(-this.moves(1,1), -this.moves(1,2), -this.moves(1,3));
 75            this.moves = this.moves(2,:, :); % pops the lase move
 76        end
 77        function success = undoAll(this)
 78            newMoves = zeros(size(this.moves,1), 3);
 79            success = 1;
 80            for n = 1:size(this.moves,1)
 81                success = success && doMove(this, -this.moves(n,1), -this.moves(n,2), -this.moves(n,3));
 82                newMoves(size(this.moves,1)-n+1, :) = [-this.moves(n,1), -this.moves(n,2), -this.moves(n,3)];
 83            end
 84        end
