1classdef APTStrainGuage < handle
 2    
 3    properties(Access = private)
 4        fig;
 5        hActiveX;
 6        serialNumber;
 7        mode;
 8    end
 9    
10    properties(Access = private, Constant)
11        CHAN1_ID = 0;
12        DISPUNITS_POSITION = 1;
13        DISPUNITS_VOLTAGE = 2;
14        DISPUNITS_FORCE = 3;
15    end
16    
17    methods(Access = public)
18        function this = APTStrainGuage(serialNumber)
19            this.fig = this.createFigure();
20            this.setSerialNumber(serialNumber);
21            this.resetControl();
22        end
23        
24        function identify(this)
25            this.hActiveX.Identify;
26            pause(1);
27            this.setMode('position');
28        end
29        
30        function setSerialNumber(this, serialNumber)
31            this.serialNumber = serialNumber;
32        end

33        
34        function resetControl(this)
35            this.hActiveX = actxcontrol('MGPIEZO.MGPiezoCtrl.1',...
36                                        [20 20 600 400], this.fig);
37            this.hActiveX.StartCtrl;
38            set(this.hActiveX,'HWSerialNum', this.serialNumber);
39        end
40        

41        function setMode(this, mode)
42            switch mode
43                case 'force'
44                    this.hActiveX.SG_SetDispMode(this.CHAN1_ID, this.DISPUNITS_FORCE);
45                    this.mode = this.DISPUNITS_FORCE;
46                case 'voltage'
47                    this.hActiveX.SG_SetDispMode(this.CHAN1_ID, this.DISPUNITS_VOLTAGE);
48                    this.mode = this.DISPUNITS_VOLTAGE;
49                case 'position'
50                    this.hActiveX.SG_SetDispMode(this.CHAN1_ID, this.DISPUNITS_POSITION);
51                    this.mode = this.DISPUNITS_POSITION;
52            end
53        end
54        
55        function pos = getPosition(this)
56            if (this.mode ~= this.DISPUNITS_POSITION)
57                this.setMode('position');
58            end
59            [channel, pos] = this.hActiveX.SG_GetReading(this.CHAN1_ID, 0);
60        end
61        function volt = getVoltage(this)
62            if (this.mode ~= this.DISPUNITS_VOLTAGE)
63                this.setMode('voltage');
64            end
65            [channel, volt] = this.hActiveX.SG_GetReading(this.CHAN1_ID, 0);
66        end
67        function force = getForce(this)
68            if (this.mode ~= this.DISPUNITS_FORCE)
69                this.setMode('force');
70            end
71            [channel, force] = this.hActiveX.SG_GetReading(this.CHAN1_ID, 0);
72        end
73    end
74    
75    methods(Access = private)
76        function fig = createFigure(this)
77            fpos    = get(0,'DefaultFigurePosition'); % figure default position
78            fpos(3) = 640; % figure window size;Width
79            fpos(4) = 480; % Height
80            
81            fig = figure('Position', fpos, 'Menu','None','Name','APT GUI');
82        end
83    end
84    
85end
